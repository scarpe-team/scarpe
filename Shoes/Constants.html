def self.find_lib_dir
  begin
    require "tmpdir"
  rescue LoadError
    return nil
  end
  homes = [
    [ENV["LOCALAPPDATA"], "Shoes"],
    [ENV["APPDATA"], "Shoes"],
    [ENV["HOME"], ".shoes"],
  ]

  top, file = homes.detect { |home_top, _| home_top && File.exist?(home_top) }
  return nil if top.nil?
  File.join(top, file)
end
